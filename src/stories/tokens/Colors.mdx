# Colors

Color palettes from the active Skeleton UI theme. Swatches and computed values update live when you switch themes or toggle dark mode in the toolbar.

export const Palette = ({ name, description }) => {
  const [, tick] = useState(0);

  useEffect(() => {
    const observer = new MutationObserver(() => tick(n => n + 1));
    observer.observe(document.documentElement, {
      attributes: true,
      attributeFilter: ['data-theme', 'class'],
    });
    return () => observer.disconnect();
  }, []);

  const steps = [50, 100, 200, 300, 400, 500, 600, 700, 800, 900, 950];

  return (
    <div style={{ marginBottom: '2.5rem' }}>
      <div style={{ display: 'flex', alignItems: 'baseline', gap: '0.625rem', marginBottom: '0.75rem' }}>
        <strong style={{ fontSize: '1rem', textTransform: 'capitalize' }}>{name}</strong>
        {description && <span style={{ color: '#6b7280', fontSize: '0.8rem' }}>{description}</span>}
      </div>
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(11, 1fr)', gap: '0.375rem' }}>
        {steps.map(step => {
          const token = `--color-${name}-${step}`;
          const value = getComputedStyle(document.documentElement).getPropertyValue(token).trim();
          return (
            <div key={step}>
              <div style={{
                height: '3rem',
                background: `var(${token})`,
                borderRadius: '0.25rem',
                border: '1px solid rgba(128,128,128,0.15)',
              }} />
              <div style={{ textAlign: 'center', marginTop: '0.3rem' }}>
                <div style={{ fontSize: '0.7rem', fontWeight: 600 }}>{step}</div>
                <div style={{ fontSize: '0.6rem', color: '#6b7280', wordBreak: 'break-all', lineHeight: 1.3, fontFamily: 'monospace' }}>
                  {value || '—'}
                </div>
              </div>
            </div>
          );
        })}
      </div>
    </div>
  );
};
